class Employee{
    //Data members
    String name;
    int age;
    long mdn;
    
    Employee()
    {
        name="KMCE";
        age=3;
        mdn=12345678;
        System.out.println("Constructor called");
    }
    
    //Parametrized Constructor
    Employee(String pname){
        System.out.println("Constructor with one param");
        this.name=pname;
    }
    
    //paramterized Constructor with 2 paramenters
    Employee(String pname,int page){
        System.out.println("Constructor with two param");
        this.name=pname;
        this.age=page;
    }
    
    //paramterized Constructor with 2 paramenters
    Employee(int page,int pmdn){
        System.out.println("Constructor with two int and long param");
        this.mdn=pmdn;
        this.age=page;
    }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        
    Employee(String pname,int page,int pmdn){
        System.out.println("Constructor with String,int and param");
        this.name=pname;
        this.mdn=pmdn;
        this.age=page;
    }
    //Methods
    //access-specifier return_type method_name(param_list){
        //}
    //Default -access
    void develop_App(){
        System.out.println("Develop App called");
    }
    
    void test_App(){
        System.out.println("Test App called");
    }
}
